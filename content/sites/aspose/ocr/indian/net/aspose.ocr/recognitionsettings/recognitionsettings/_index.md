public RecognitionSettings(Language language = Language.None, 
    List<Rectangle> recognitionAreas = null, bool detectAreas = true, bool autoSkew = true,

| पैरामीटर | प्रकार | विवरण |
| --- | --- | --- |
| language | Language | ओसीआर के लिए प्रयुक्त भाषा। |

| detectAreas | Boolean | स्वचालित पाठ क्षेत्रों का पता लगाना सक्षम करें। |
| autoSkew | Boolean | स्वचालित छवि तिरछा सुधार सक्षम करें। |
| skewAngle | Single | छवि रोटेशन के लिए कोण सेट करें। |
| recognizeSingleLine | Boolean | सिंगल-लाइन छवियों के लिए |
| threshold | Int32 | कस्टम इमेज बाइनराइजेशन थ्रेशोल्ड |
